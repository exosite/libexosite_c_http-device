118 EXO_STATE exosite_init(const char * vendor, const char *model)
119 {
120  uint8_t retStatus = 0;
121  // reset state
122  initState = EXO_STATE_NOT_COMPLETE;
123 
124  exoPal_init();
125 
126  // get cik and uuid and any other nvm stored data, into ram.
127  retStatus |= exoPal_getCik(cikBuffer);
128  retStatus |= exoPal_getUuid(uuidBuffer);
129  exoPal_memcpy(vendorBuffer, vendor, MAX_VENDOR_LENGTH);
130  exoPal_memcpy(modelBuffer, model, MAX_MODEL_LENGTH);

146 uint8_t exoPal_getResponse(char * buffer, uint16_t bufferSize, uint16_t * responseLength)
147 {
148  int32_t response;
149  uint16_t length = 0;
150  uint8_t results = 0;
151  char *bodyStart = 0;
152  uint16_t bodyLength = 0;
153  int32_t readAttempts = 0;
154 
155 
156  // read from socket. This assumes that the first read attempt will return
157  // at a minimum the content length header
158  results = exoPal_socketRead(buffer, bufferSize, responseLength);
159 
160  if (results != 0)
161  {
162  return results;
163  }
164 
165  // find the start of the body.
166  response = exosite_getBody(buffer, &bodyStart, &bodyLength);
167 
168  if (response != 0)
169  {
170  return response;
171  }
172 
173  // while received data < expected data as specified by the content header
174  while ((*responseLength - (bodyStart - buffer)) < bodyLength)
175  {
176 
177  results = exoPal_socketRead(buffer + *responseLength, bufferSize, &length);
178  if (length == 0)
179  {
180  // this is to prevent a dangling read from hanging up the system for
181  // an extended period of time.
182  readAttempts++;
183  if (readAttempts >= MAX_RESPONSE_READ_ATTEMPTS)
184  {
185  return 20;
186  }
187  }
188  if (results != 0)
189  {
190  return results;
191  }
192  *responseLength += length;
193  }
194 
195  return 0;
196 }

206 EXO_STATE exosite_activate()
207 {
208  // assume content length will never be over 9999 bytes
209  char contentLengthStr[5];
210  uint8_t len_of_contentLengthStr;
211  EXO_STATE retVal;
212  uint16_t responseLen;
213  uint8_t connection_status;
214 
215 
216  // Try and activate device with Exosite, four possible cases:
217  // * We don't have a stored CIK and receive a 200 response with a CIK
218  // * Means device was enabled and this was our first connection
219  // * We don't have a stored CIK and receive a 409 response
220  // * The device is not enabled.
221  // * We have a stored CIK and receive a 409 response.
222  // * Device has already been activated and has a valid CIK
223  // * We have a stored CIK and receive a 401 response
224  // * R/W error
225 
226  uint8_t vendorLength = exoPal_strlen(vendorBuffer);
227  uint8_t modelLength = exoPal_strlen(modelBuffer);
228  uint8_t uuidLength = exoPal_strlen(uuidBuffer);
229 
230  // get body length
231  uint16_t bodyLength = sizeof(STR_VENDOR) - 1 +
232  sizeof(STR_MODEL) - 1 +
233  sizeof(STR_SN) - 1;
234  bodyLength += vendorLength + modelLength + uuidLength;
235 
236 
237  len_of_contentLengthStr = exoPal_itoa((int)bodyLength, contentLengthStr, 5);
238 
239 
240  connection_status = exosite_connect();
241 
242  // return error message if connect failed.
243  if (connection_status != 0)
244  {
245  return EXO_STATE_CONNECTION_ERROR;
246  }
247 
248  // send request
249  exoPal_socketWrite(STR_ACTIVATE_URL, sizeof(STR_ACTIVATE_URL) - 1);
250  exoPal_socketWrite(STR_HTTP, sizeof(STR_HTTP) - 1);
251  exoPal_socketWrite(STR_CRLF, sizeof(STR_CRLF) - 1);
252 
253  // send Host header
254  exoPal_socketWrite(STR_HOST, sizeof(STR_HOST) - 1);
255  exoPal_socketWrite(STR_CRLF, sizeof(STR_CRLF) - 1);
256 
257  // send content type header
258  exoPal_socketWrite(STR_CONTENT, sizeof(STR_CONTENT) - 1);
259  exoPal_socketWrite(STR_CRLF, sizeof(STR_CRLF) - 1);
260 
261  // send content length header
262  exoPal_socketWrite(STR_CONTENT_LENGTH, sizeof(STR_CONTENT_LENGTH) - 1);
263  exoPal_socketWrite(contentLengthStr, len_of_contentLengthStr);
264  exoPal_socketWrite(STR_CRLF, sizeof(STR_CRLF) - 1);
265  exoPal_socketWrite(STR_CRLF, sizeof(STR_CRLF) - 1);
266 
267  // send body
268  exoPal_socketWrite(STR_VENDOR, sizeof(STR_VENDOR) - 1);
269  exoPal_socketWrite(vendorBuffer, vendorLength);
270  exoPal_socketWrite(STR_MODEL, sizeof(STR_MODEL) - 1);
271  exoPal_socketWrite(modelBuffer, modelLength);
272  exoPal_socketWrite(STR_SN, sizeof(STR_SN) - 1);
273  exoPal_socketWrite(uuidBuffer, uuidLength);
274 
275  exoPal_sendingComplete();
276 
277  retVal = EXO_STATE_CONNECTION_ERROR;
278 
279 
280 
281 
282  exoPal_getResponse(exoPal_rxBuffer, RX_BUFFER_SIZE, &responseLen);
283 
284  exosite_disconnect();
285 
286 
287  if (responseLen == 0)
288  {
289  // if we didn't receive any data from the modem
290  retVal = EXO_STATE_NO_RESPONSE;
291  }
292  else if (exosite_checkResponse(exoPal_rxBuffer, "200"))
293  {
294  // we received a CIK.
295  char * bodyStart;
296  uint16_t bodyLength = 0;
297  int32_t response;
298  response = exosite_getBody(exoPal_rxBuffer, &bodyStart, &bodyLength);
299  if (response < 0)
300  {
301  // some error getting the body
302  return EXO_STATE_CONNECTION_ERROR;
303  }
304  if ((exosite_isCIKValid(bodyStart)) && (bodyLength == CIK_LENGTH))
305  {
306  // got a valid cik in the response
307  exoPal_setCik(bodyStart);
308  exoPal_memcpy(cikBuffer, bodyStart, CIK_LENGTH);
309  retVal = EXO_STATE_VALID_CIK;
310  }
311  }
312  else if (exosite_checkResponse(exoPal_rxBuffer, "409"))
313  {
314  exoPal_getCik(cikBuffer);
315 
316  if (exosite_isCIKValid(cikBuffer))
317  {
318  // If we receive a 409 and we do have a valid CIK, we will
319  // assume we are good to go.
320  retVal = EXO_STATE_VALID_CIK;
321  }
322  else
323  {
324  // if we don't have a CIK in nvm and we receive a 409
325  // The device isn't enabled in the dashboard
326  retVal = EXO_STATE_DEVICE_NOT_ENABLED;
327  }
328  }
329  else if (exosite_checkResponse(exoPal_rxBuffer, "401"))
330  {
331  // RW error
332  retVal = EXO_STATE_R_W_ERROR;
333  }
334 
335  return retVal;
336 
337 }

349 int32_t exosite_getBody(char *response, char **bodyStart, uint16_t *bodyLength)
350 {
351  // find content length
352  char* strStart;
353  char* charAfterContentLengthValue = 0;
354  char cr[] = "\r";
355  char httpBodyToken[] = "\r\n\r\n";
356  // find start of content length header
357  strStart = exoPal_strstr(response, STR_CONTENT_LENGTH);
358  if (strStart <= 0)
359  {
360  return -1;
361  }
362  strStart = strStart + sizeof(STR_CONTENT_LENGTH) - 1;
363  // get \r and set to '\0' for atoi
364  charAfterContentLengthValue = exoPal_strstr(strStart, cr);
365  if (charAfterContentLengthValue <= 0)
366  {
367  return -2;
368  }
369  if (charAfterContentLengthValue != 0)
370  {
371  // temporarily null terminate the content length
372  *charAfterContentLengthValue = '\0';
373  *bodyLength = exoPal_atoi(strStart);
374  *charAfterContentLengthValue = '\r';
375 
376 
377  strStart = exoPal_strstr(strStart, httpBodyToken);
378  if (strStart <= 0)
379  {
380  return -3;
381  }
382  *bodyStart = (char *)(strStart + sizeof(httpBodyToken)-1);
383  }
384  return 0;
385 }
