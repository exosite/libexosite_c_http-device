86 void exosite_send_http_req(Exosite_state_t *state)
87 {
88  size_t slen = 0;
89  exoHttp_req_t *req = &state->http.req;
90  switch(req->step) {
91  case exoHttp_req_method_url:
92  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
93  if(req->is_post) {
94  exoPal_strlcpy(state->workbuf, STR_POST, sizeof(state->workbuf));
95  } else {
96  exoPal_strlcpy(state->workbuf, STR_GET, sizeof(state->workbuf));
97  }
98  slen = exoPal_strlcat(state->workbuf, req->method_url_path, sizeof(state->workbuf));
99 
100  if(req->query != NULL) {
101  req->step = exoHttp_req_query;
102  slen = exoPal_strlcat(state->workbuf, STR_QM, sizeof(state->workbuf));
103  } else {
104  req->step = exoHttp_req_http_version;
105  }
106  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
107  break;
108 
109  case exoHttp_req_query:
110  req->step = exoHttp_req_http_version;
111  exoPal_socketWrite(state->exoPal, req->query, exoPal_strlen(req->query));
112  break;
113 
114  case exoHttp_req_http_version:
115  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
116  exoPal_strlcpy(state->workbuf, STR_HTTP, sizeof(state->workbuf));
117  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
118 
119  req->step = exoHttp_req_host;
120  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
121  break;
122 
123  case exoHttp_req_host:
124  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
125  slen = exoPal_strlcpy(state->workbuf, STR_HOST, sizeof(state->workbuf));
126  exoPal_getHostname(&state->workbuf[slen], sizeof(state->workbuf)-slen);
127  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
128 
129  req->step = exoHttp_req_agent;
130  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
131  break;
132 
133  case exoHttp_req_agent:
134  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
135  exoPal_strlcpy(state->workbuf, STR_AGENT_HEADER, sizeof(state->workbuf));
136  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
137 
138  req->step = exoHttp_req_connection;
139  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
140  break;
141 
142  case exoHttp_req_connection:
143  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
144  exoPal_strlcpy(state->workbuf, STR_CONNECTION_CLOSE, sizeof(state->workbuf));
145  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
146 
147  req->step = exoHttp_req_cik;
148  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
149  break;
150 
151 
152  case exoHttp_req_cik:
153  req->step = exoHttp_req_content;
154  // FIXME: If using TLS Client Certificate, then NEVER send this header.
155  // Or if state->cik[0] == 0 ? // && strlen(state->cik) == 40 ???
156  if(req->include_cik) {
157  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
158  exoPal_strlcpy(state->workbuf, STR_CIK_HEADER, sizeof(state->workbuf));
159  exoPal_strlcat(state->workbuf, state->cik, sizeof(state->workbuf));
160  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
161 
162  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
163  break;
164  }
165  // Fall-thru if no CIK
166 
167  case exoHttp_req_content:
168  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
169  exoPal_strlcpy(state->workbuf, STR_CONTENT, sizeof(state->workbuf));
170  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
171 
172  req->step = exoHttp_req_accept;
173  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
174  break;
175 
176  case exoHttp_req_accept:
177  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
178  exoPal_strlcpy(state->workbuf, STR_ACCEPT, sizeof(state->workbuf));
179  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
180 
181  req->step = exoHttp_req_timeout;
182  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
183  break;
184 
185  case exoHttp_req_timeout:
186  req->step = exoHttp_req_modified_since;
187  if(req->request_timeout > 0) {
188  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
189  slen = exoPal_strlcpy(state->workbuf, STR_REQUEST_TIMEOUT, sizeof(state->workbuf));
190  exoPal_itoa(req->request_timeout, &state->workbuf[slen], sizeof(state->workbuf)-slen);
191  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
192 
193  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
194  break;
195  }
196  // Fall-thru if no timeout
197 
198  case exoHttp_req_modified_since:
199  req->step = exoHttp_req_content_length;
200  if(req->modified_since != NULL) {
201  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
202  exoPal_strlcpy(state->workbuf, STR_MODIFIED_SINCE, sizeof(state->workbuf));
203  exoPal_strlcat(state->workbuf, req->modified_since, sizeof(state->workbuf));
204  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
205 
206  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
207  break;
208  }
209  // Fall-thru if no modified_since
210 
211  case exoHttp_req_content_length:
212  req->step = exoHttp_req_start_body;
213  if(req->content_length > 0) {
214  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
215  slen = exoPal_strlcpy(state->workbuf, STR_CONTENT_LENGTH, sizeof(state->workbuf));
216  exoPal_itoa(req->content_length, &state->workbuf[slen], sizeof(state->workbuf)-slen);
217  slen = exoPal_strlcat(state->workbuf, STR_CRLF, sizeof(state->workbuf));
218 
219  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
220  break;
221  }
222  // Fall-thru if no content_length
223 
224  case exoHttp_req_start_body:
225  req->step = exoHttp_req_body;
226  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
227  slen = exoPal_strlcpy(state->workbuf, STR_CRLF, sizeof(state->workbuf));
228  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
229  break;
230 
231  case exoHttp_req_body:
232  req->step = exoHttp_req_complete;
233  if(req->is_activate) {
234  exoPal_memset(state->workbuf, 0, sizeof(state->workbuf));
235  exoPal_strlcpy(state->workbuf, STR_VENDOR, sizeof(state->workbuf));
236  exoPal_strlcat(state->workbuf, state->projectid, sizeof(state->workbuf));
237  exoPal_strlcat(state->workbuf, STR_MODEL, sizeof(state->workbuf));
238  exoPal_strlcat(state->workbuf, state->modelid, sizeof(state->workbuf));
239  exoPal_strlcat(state->workbuf, STR_SN, sizeof(state->workbuf));
240  slen = exoPal_strlcat(state->workbuf, state->uuid, sizeof(state->workbuf));
241 
242  exoPal_socketWrite(state->exoPal, state->workbuf, slen);
243  break;
244  } else if(req->body != NULL) {
245  exoPal_socketWrite(state->exoPal, req->body, exoPal_strlen(req->body));
246  break;
247  }
248  // Fall-thru if no body
249 
250  case exoHttp_req_complete:
251  exoPal_sendingComplete(state->exoPal);
252  state->stage = Exosite_Stage_waiting;
253  break;
254  }

279 void exosite_http_rpl_parse(Exosite_state_t *state, const char *data, size_t len)
280 {
281  exoHttp_rpl_t *rpl = &state->http.rpl;
282  char dc;
283  const char *start_mark = data;
284 
285  // Special case speed up. If in the exoHttp_rpl_body, nothing to scan.
286  if(rpl->step == exoHttp_rpl_body) {
287  rpl->counter_content_length -= len;
288  if(exosite_http_rpl_body(state, data, len) != 0) {
289  rpl->step = exoHttp_rpl_complete;
290  }
291  if(rpl->counter_content_length <= 0) {
292  rpl->step = exoHttp_rpl_complete;
293  }
294  return;
295  }
296 
297  for(; len > 0; ++data, --len) {
298  dc = *data;
299  switch(rpl->step) {
300  case exoHttp_rpl_looking_for_status:
301  if(dc == ' ') {
302  rpl->step = exoHttp_rpl_read_status;
303  }
304  break;
305  case exoHttp_rpl_read_status:
306  if(dc >= '0' && dc <= '9') {
307  rpl->statusCode *= 10;
308  rpl->statusCode += dc - '0';
309  } else if(dc == ' ') {
310  rpl->step = exoHttp_rpl_status_looking_for_cr;
311  } else {
312  rpl->step = exoHttp_rpl_error;
313  }
314  break;
315  case exoHttp_rpl_status_looking_for_cr:
316  if(dc == '\r') {
317  rpl->step = exoHttp_rpl_status_looking_for_lf;
318  }
319  break;
320  case exoHttp_rpl_status_looking_for_lf:
321  if(dc != '\n') {
322  rpl->step = exoHttp_rpl_error;
323  } else {
324  rpl->step = exoHttp_rpl_header_start;
325  }
326  break;
327  case exoHttp_rpl_header_start:
328  if(dc == '\r') {
329  // This is either start of second CRLF to start body, or will be
330  // error.
331  rpl->step = exoHttp_rpl_looking_for_lf_start_body;
332  } else {
333  // Watch for Content-Length.
334  if(exosite_to_lower(dc) == 'c') {
335  rpl->counter_content_length = 1;
336  } else {
337  rpl->counter_content_length = -1;
338  }
339  start_mark = data;
340  rpl->step = exoHttp_rpl_header_looking_for_sep;
341  }
342  break;
343  case exoHttp_rpl_header_looking_for_sep:
344  if(dc == ':') {
345  // switch to value
346  rpl->step = exoHttp_rpl_header_mark_value;
347  exosite_http_rpl_header_name(state, start_mark, data-start_mark);
348  } else if(rpl->counter_content_length >= 0) {
349  // Still Content-Length?
350  if( exosite_to_lower(dc) == exosite_to_lower(STR_CONTENT_LENGTH[rpl->counter_content_length])) {
351  rpl->counter_content_length ++;
352  } else {
353  rpl->counter_content_length = -1;
354  }
355  }
356  break;
357  case exoHttp_rpl_header_mark_value:
358  start_mark = data;
359  rpl->step = exoHttp_rpl_header_looking_for_cr;
360  if(rpl->counter_content_length > 0) {
361  rpl->content_length = 0;
362  }
363  // this *must* fall-thru
364  case exoHttp_rpl_header_looking_for_cr:
365  if(dc == '\r') {
366  rpl->step = exoHttp_rpl_header_looking_for_lf;
367  exosite_http_rpl_header_value(state, start_mark, data-start_mark);
368  } else if(rpl->counter_content_length > 0) {
369  // This is Content-Length
370  if(dc >= '0' && dc <= '9') {
371  rpl->content_length *= 10;
372  rpl->content_length += dc - '0';
373  }
374  }
375  break;
376  case exoHttp_rpl_header_looking_for_lf:
377  if(dc == '\n') {
378  // end of header.
379  // Next header, or maybe body.
380  rpl->step = exoHttp_rpl_header_start;
381  } else {
382  rpl->step = exoHttp_rpl_error;
383  }
384  rpl->counter_content_length = -1;
385  break;
386 
387 
388  case exoHttp_rpl_looking_for_lf_start_body:
389  if(dc != '\n') {
390  rpl->step = exoHttp_rpl_error;
391  } else {
392  rpl->step = exoHttp_rpl_mark_body;
393  }
394  break;
395  case exoHttp_rpl_mark_body:
396  rpl->step = exoHttp_rpl_body;
397  start_mark = data;
398  if(rpl->content_length > 0) {
399  rpl->counter_content_length = rpl->content_length;
400  } else {
401  rpl->counter_content_length = INT_MAX;
402  }
403  // this *must* fall-thru
404  case exoHttp_rpl_body:
405  // just get to the end now.
406  break;
407 
408  case exoHttp_rpl_complete:
409  case exoHttp_rpl_error:
410  break;
411  }
412  }
413  // Handle partial callbacks.
414  switch(rpl->step) {
415  case exoHttp_rpl_header_looking_for_sep:
416  exosite_http_rpl_header_name(state, start_mark, data-start_mark);
417  break;
418 
419  case exoHttp_rpl_header_looking_for_cr:
420  exosite_http_rpl_header_value(state, start_mark, data-start_mark);
421  break;
422 
423  case exoHttp_rpl_body:
424  rpl->counter_content_length -= data-start_mark;
425  if(exosite_http_rpl_body(state, start_mark, data-start_mark) != 0) {
426  rpl->step = exoHttp_rpl_complete;
427  }
428  if(rpl->counter_content_length <= 0) {
429  rpl->step = exoHttp_rpl_complete;
430  }
431  break;
432 
433  default:
434  // nop.
435  break;
436  }

441 void exosite_activate(Exosite_state_t *state)
442 {
443  state->state = Exosite_State_activate;
444  state->stage = Exosite_Stage_connecting;
445 
446  state->http.req.step = exoHttp_req_method_url;
447  state->http.req.method_url_path = (char*)STR_ACTIVATE_URL;
448  state->http.req.content_length = (sizeof(STR_VENDOR) - 1) +
449  (sizeof(STR_MODEL) - 1) + (sizeof(STR_SN) - 1) +
450  exoPal_strlen(state->projectid) +
451  exoPal_strlen(state->modelid) +
452  exoPal_strlen(state->uuid);
453  state->http.req.is_post = 1;
454  state->http.req.include_cik = 0;
455  state->http.req.is_activate = 1;
456  state->http.req.body = NULL;
457  state->http.req.query = NULL;
458  state->http.req.modified_since = NULL;
459  state->http.req.request_timeout = 0;
460 
461  exoPal_tcpSocketOpen(state->exoPal);
462 }

541 int exosite_lib_recv(exoPal_state_t *pal, const char *data, size_t len)
542 {
543  Exosite_state_t *state = (Exosite_state_t *)pal->context;
544  EXO_ASSERT(pal->context);
545 
546  // len == 0 means EOS, so close socket.
547  if(len == 0) {
548  state->stage = Exosite_Stage_closing;
549  exoPal_tcpSocketClose(state->exoPal);
550  return 0;
551  }
552 
553  exosite_http_rpl_parse(state, data, len);
554 
555  if(state->http.rpl.step == exoHttp_rpl_complete) {
556  state->stage = Exosite_Stage_closing;
557  exoPal_tcpSocketClose(state->exoPal);
558 
559  } else if(state->http.rpl.step == exoHttp_rpl_error) {
560  state->stage = Exosite_Stage_closing;
561  exoPal_tcpSocketClose(state->exoPal);
562  // TODO: callback?
563 
564  } else {
565  exoPal_socketRead(state->exoPal, state->workbuf, sizeof(state->workbuf));
566  }
567  return 0;

571 int exosite_http_rpl_body(Exosite_state_t *state, const char *data, size_t len)
572 {
573  int retcode = 0;
574  if(state->stage == Exosite_Stage_recving) {
575  state->stage = Exosite_Stage_recving_body;
576 
577  // Fire off read_begin callbacks.
578  switch(state->state) {
579  case Exosite_State_read:
580  case Exosite_State_hybrid:
581  case Exosite_State_longpoll:
582  if(state->ops.on_read_begin) {
583  state->ops.on_read_begin(state, state->http.rpl.statusCode);
584  }
585  break;
586  default:
587  // nop
588  break;
589  }
590  }
591 
592  // Process body data.
593  switch(state->state) {
594  case Exosite_State_activate:
595  if(state->http.rpl.statusCode == 409) {
596  // Already activated.
597  exoPal_getCik(state->cik, CIK_LENGTH);
598  if(!exosite_isCIKValid(state->cik)) {
599  // Stored CIK is invalid and this SN is activated.
600  state->statusCode = -409;
601  }
602 
603  // activate complete!
604  state->statusCode = 200;
605  retcode = 1;
606 
607  } else if(state->http.rpl.statusCode == 200) {
608  // Just activated.
609 
610  // copy into cik, by parts if need be.
611  unsigned int remain = MIN(len, (unsigned int)(CIK_LENGTH - state->wb_offset));
612  EXO_ASSERT(remain > 0);
613  exoPal_memmove(&state->cik[state->wb_offset], data, remain);
614  state->wb_offset += remain;
615  EXO_ASSERT(state->wb_offset <= CIK_LENGTH);
616 
617  if(state->wb_offset == CIK_LENGTH) {
618  // ok, all of CIK is now in cik; save it.
619  if(exosite_isCIKValid(state->cik)) {
620  exoPal_setCik(state->cik, CIK_LENGTH);
621  state->statusCode = 200;
622  } else {
623  // It gave us an invalid CIK!?!?!?!
624  state->statusCode = -200;
625  }
626 
627  retcode = 1;
628  }
629  } else if(state->http.rpl.statusCode == 415) {
630  // We identified with a TLS Client Certificate.
631  // Carry on.
632  state->statusCode = 200;
633  retcode = 1;
634 
635  } else {
636  // All other status codes.
637  state->statusCode = state->http.rpl.statusCode;
638  retcode = 1;
639  }
640  break;
641 
642  case Exosite_State_write:
643  // If they send back a body, we ignore it.
644  retcode = 1;
645  break;
646 
647  case Exosite_State_read:
648  case Exosite_State_hybrid:
649  case Exosite_State_longpoll:
650  if(state->http.rpl.statusCode != 200) {
651  // just stop.
652  retcode = 1;
653 
654  } else {
655  if(state->ops.on_read_raw) {
656  state->ops.on_read_raw(state, data, len);
657  }
658  // TODO: decode form data.
659  // Do this in here? or provide a layer above?
660  }
661  break;
662 
663  case Exosite_State_timestamp:
664  if(state->http.rpl.statusCode != 200) {
665  // just stop.
666  retcode = 1;
667 
668  } else {
669  // The actual timestamp values could be broken across multiple
670  // calls of this function.
671 
672  for(; len > 0; ++data, --len) {
673  char dc = *data;
674  if(dc >= '0' && dc <= '9') {
675  state->timestamp *= 10;
676  state->timestamp += dc - '0';
677  } else {
678  // Not number, stop.
679  retcode = 1;
680  break;
681  }
682  }
683  }
684  break;
685 
686  default:
687  EXO_ASSERT(0);
688  break;
689  }
690  return retcode;
691 }
